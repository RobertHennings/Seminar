\subsection{Research Hypothesis}
%----------------------------------------------------------------------------------------
%	NEW SLIDE
%----------------------------------------------------------------------------------------
\begin{frame}
	\frametitle{Formulated Research Hypothesis}
	\scriptsize
	\begin{block}{\underline{Main Research Hypothesis}}
		\centering
		\textit{``The standard UIP-equilibrium condition is \textbf{time-dependent} and primarily controlled by \textbf{two main regimes}, characterised by either \textbf{high or low (market-) volatility}.''}
	\end{block}
	\begin{block}{Additional Research Hypothesis I}
		\centering
		\textit{``Monetary policy, i.e. \textbf{interest rates}, is partly driven by \textbf{energy commodity prices} that induce volatility through the inflationary pass-through channel, especially during \textbf{phases of market distress} in economies heavily relying on import/export of energy commodities.''}
	\end{block}
	\begin{block}{Additional Research Hypothesis II}
		\centering
		\textit{``Factoring in variables related to \textbf{energy commodity prices} in combination with using \textbf{alternative clustering techniques} improves the \textbf{identification of the regimes} to better pinpoint the time-dependent testing of the standard UIP relation, compared to Markov-Switching benchmark models.''}
	\end{block}
\end{frame}
