\begin{frame}
    \frametitle{Appendix - Definitions}
    
    \begin{block}{PPP Deviation Calculation}
        \begin{equation*}
            \text{PPP Deviation} = \frac{\text{Actual Exchange Rate} - \text{PPP Exchange Rate}}{\text{PPP Exchange Rate}} \times 100
        \end{equation*}
        \footnotesize{Source: \cite{oecd_2025_purchasing_power_parities}}
    \end{block}
    
    \begin{block}{Clustering Metrics: Silhouette Score}
        \begin{equation*}
            \text{Silhouette Score} = \frac{b - a}{\max\left(a, b\right)}
        \end{equation*}
        \footnotesize{Source: \url{https://scikit-learn.org/stable/modules/generated/sklearn.metrics.silhouette_score.html}}
    \end{block}
\end{frame}

\begin{frame}
	\frametitle{Appendix - Definitions}
    \underline{Calculation of main exchange rate types (Slide XX):}
    \begin{itemize}
        \item Nominal effective exchange rate (NEER): Calculated as geometric trade-weighted averages of bilateral exchange rates.\footnote{Definition taken from \cite{bis_2025_effective_exchange_rates}.}
        \item Real effective exchange rates (REER): Derived by adjusting the NEER by relative consumer prices.\footnote{Definition taken from \cite{bis_2025_effective_exchange_rates}.}
        \item Nominal exchange rates (NER): The exchange rate between two currencies without adjustment for inflation.\footnote{Definition taken from \cite{ca_zorzi_2018_exchange_rate_forecasting}, page 5.}
        \item Real exchange rates (RER): The nominal exchange rate adjusted for differences in price levels between countries.\footnote{Definition taken from \cite{ca_zorzi_2018_exchange_rate_forecasting}, page 5.}
    \end{itemize}
\end{frame}
