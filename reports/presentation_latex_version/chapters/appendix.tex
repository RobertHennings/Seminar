\begin{frame}
	\frametitle{Appendix - Definitions}
    % \underline{Calculation of deviations of USD Spot Rate from PPP-values (Slide 04):}
    \begin{block}{PPP Deviation Calculation}
        \begin{equation}
            \text{PPP Deviation} = \log(REER_{country} - REER_{USD})
        \end{equation}
        \footnotesize{Source: \cite{oecd_2025_purchasing_power_parities}, data taken from \cite{fred_2025_federal_reserve_data}, last accessed 24.10.25.}
    \end{block}

    % \underline{Calculation of component distributions of the CPI (Slide 06, 07):}
    \begin{block}{CPI Component Distribution Calculation}
        \begin{equation}
            \text{CPI Component Distribution} = \frac{\text{CPI}_{component}}{\text{CPI}_{total}} \times 100
        \end{equation}
        \footnotesize{Source: Data taken from \cite{fred_2025_federal_reserve_data} and \cite{ecb_2025_data_portal}, last accessed 24.10.25.}
    \end{block}
\end{frame}

    % \underline{Calculation of rolling (30-day) volatility (Slide 08):}
    \begin{block}{Rolling Volatility Calculation}
        \begin{equation}
            \text{Rolling Volatility} = \sqrt{\frac{1}{N} \sum_{t=1}^{N} (x_t - \bar{x})^2}
        \end{equation}
        \footnotesize{Source: Data taken from \cite{fred_2025_federal_reserve_data}, last accessed 24.10.25.}
    \end{block}

\begin{frame}
    \frametitle{Appendix - Definitions}

    % \underline{Calculation of main exchange rate types (Slide 13):}

    \begin{block}{Nominal effective exchange rate (NEER) (Slide 13):}
        \begin{equation}
            \text{NEER} = \frac{1}{N} \sum_{i=1}^{N} \text{NER}_{i}
        \end{equation}
        \footnotesize{Source: Definition taken from \cite{bis_2025_effective_exchange_rates}.}
    \end{block}

    \begin{block}{Real effective exchange rate (REER) (Slide 13):}
        \begin{equation}
            \text{REER} = \text{NEER} \times \frac{\text{CPI}_{domestic}}{\text{CPI}_{foreign}}
        \end{equation}
        \footnotesize{Source: Definition taken from \cite{bis_2025_effective_exchange_rates}.}
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{Appendix - Definitions}

    \begin{block}{Nominal exchange rate (NER) (Slide 13):}
        \begin{equation}
            \text{NER} = \frac{\text{Units of Domestic Currency}}{\text{Units of Foreign Currency}}
        \end{equation}
        \footnotesize{Source: Definition taken from \cite{ca_zorzi_2018_exchange_rate_forecasting}, page 5.}
    \end{block}

    \begin{block}{Real exchange rate (RER) (Slide 13):}
        \begin{equation}
            \text{RER} = \text{NER} \times \frac{\text{CPI}_{domestic}}{\text{CPI}_{foreign}}
        \end{equation}
        \footnotesize{Source: Definition taken from \cite{ca_zorzi_2018_exchange_rate_forecasting}, page 5.}
    \end{block}
    
    \begin{block}{Clustering Metrics: Silhouette Score (Slide 17):}
        \begin{equation}
            \text{Silhouette Score} = \frac{b - a}{\max\left(a, b\right)}
        \end{equation}
        \footnotesize{Source: Definition taken from \cite{sklearn_silhouette_score}.}
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{Appendix - Definitions}

    \begin{block}{Standard UIP relationship - log terms (Slide XX):}
        \begin{equation}
            \bigtriangleup s_{t + 1} = i_{t} - i_{t}^{*} + u_{t+1}
        \end{equation}
        \footnotesize{Source: Definition taken from \cite{reitz_2025_applied_econometrics_fx}, section "Testing UIP conditions", page 7.}
    \end{block}

    \begin{block}{Calculation of the RCM (Slide 17):}
        \begin{equation}
            \text{RCM} = 100\,S^{2}\left(1-\frac{1}{N}\sum_{t=1}^{N}\big[g_t p_t + (1-g_t)(1-p_t)\big]\right)
        \end{equation}
        \footnotesize{with \(g_t=\mathbf{1}\{p_t\ge\tau\}\), \(p_t\) the smoothed probability at time \(t\), \(N\) the sample size, and \(\tau\) the classification threshold (e.g. 0.5).}
        \footnotesize{Source: Definition taken from \cite{ang1998regime}.}
    \end{block}
